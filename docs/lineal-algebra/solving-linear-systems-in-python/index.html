import numpy as np

# Coefficient matrix (3x3)
A = np.array([[2, 1, -1],
              [-3, -1, 2],
              [-2, 1, 2]], dtype=float)

# Constants vector (1D is common for np.linalg.solve)

# Solve the system A * x = b
x = np.linalg.solve(A, b)

print("Solution:")
print(x)
